📦 Tutoriels d'installation – Crimson Domination

────────────────────────────

☕ INSTALLER JAVA

sudo apt update && sudo apt install -y openjdk-17-jre
────────────────────────────

📁 CRÉER LES DOSSIERS POUR LES SERVEURS

mkdir -p /home/serveurs/{bungee,lobby,minage,survie1,survie2,site}
────────────────────────────

⬇️ TÉLÉCHARGER LES SERVEURS MINECRAFT (ex. Paper)

cd /home/serveurs/lobby
curl -o paper.jar https://api.papermc.io/v2/projects/paper/versions/1.20.4/builds/550/downloads/paper-1.20.4-550.jar
────────────────────────────

🚀 LANCER LE SERVEUR AVEC RAM DÉFINIE

java -Xmx3G -jar paper.jar --nogui
────────────────────────────

🔗 INSTALLER WATERFALL (RECOMMANDÉ POUR LE PROXY)

cd /home/serveurs/bungee
curl -o waterfall.jar https://papermc.io/api/v2/projects/waterfall/versions/1.20/builds/539/downloads/waterfall-1.20-539.jar
java -Xmx1G -jar waterfall.jar
────────────────────────────

🐳 INSTALLER DOCKER + PORTAINER

sudo apt install docker.io -y
sudo systemctl enable docker

sudo docker volume create portainer_data

sudo docker run -d -p 9443:9443 \
  --name portainer \
  --restart=always \
  -v /var/run/docker.sock:/var/run/docker.sock \
  -v portainer_data:/data \
  portainer/portainer-ce:latest
────────────────────────────

🌐 INSTALLER APACHE + MARIADB + PHP + PHPMYADMIN

sudo apt install apache2 mariadb-server php php-mysql libapache2-mod-php phpmyadmin
────────────────────────────

💻 INSTALLER AZURIOM (SITE WEB)

cd /var/www/html
sudo wget https://azuriom.com/download -O azuriom.zip
sudo unzip azuriom.zip
sudo chown -R www-data:www-data /var/www/html
